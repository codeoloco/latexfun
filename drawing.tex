% examples from this post
% http://timmurphy.org/2010/08/20/drawing-pictures-in-latex/

\documentclass{article}
\begin{document}

\setlength{\unitlength}{1cm}

% circle
\begin{picture}(2,2)
\put(1,1){\circle{1}}
\end{picture}

% lines
\begin{picture}(5, 5)
\multiput(0,0)(0.5,1){5}{\line(1,0){0.5}}
\end{picture}

%shortstack
\begin{picture}(5,5)
\shortstack{\circle{1}\\\circle{1}\\\circle{1}}
\end{picture}

% available shapes

% outlined circle
\begin{picture}(2,2)
\put(1,1){\circle{1}}
\end{picture}

% filled circle
\begin{picture}(2,2)
\put(1,1){\circle*{1}}
\end{picture}

% dashbox
\begin{picture}(2,2)
\put(0,0){\dashbox{0.2}(2,2)[s]{Boo}}
\end{picture}

% frame
\begin{picture}(2,2)
\put(1,1){\frame{Boo}}
\end{picture}

% framebox
\begin{picture}(2,2)
\put(0,0){\framebox(2,2){Boo}}
\end{picture}

% line
\begin{picture}(3,1)
\put(0,0){\line(3,1){3}}
\end{picture}

% makebox
\begin{picture}(2,2)
\put(0,0){\makebox{Boo}}
\end{picture}

% oval
\begin{picture}(2,2)
\put(1,1){\oval(2,1)}
\end{picture}

% vector
\begin{picture}(3,1)
\put(0,0){\vector(3,1){3}}
\end{picture}

\end{document}
